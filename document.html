// Superclass class PaymentProcessor { processPayment(amount) { throw new Error("Este método debe ser sobrescrito"); } } // Concrete Processors class CreditCardProcessor extends PaymentProcessor { processPayment(amount) { console.log(`Procesando pago de $${amount} con tarjeta de crédito.`); } } class PayPalProcessor extends PaymentProcessor { processPayment(amount) { console.log(`Procesando pago de $${amount} con PayPal.`); } } class CryptoProcessor extends PaymentProcessor { processPayment(amount) { console.log(`Procesando pago de $${amount} con criptomoneda.`); } } // Factory Method function getPaymentProcessor(type) { switch(type) { case 'creditCard': return new CreditCardProcessor(); case 'paypal': return new PayPalProcessor(); case 'crypto': return new CryptoProcessor(); default: throw new Error("Método de pago no soportado"); } } // Client Code const paymentType = 'paypal'; // Puede venir de la elección del usuario const processor = getPaymentProcessor(paymentType); processor.processPayment(100); // Procesando pago de $100 con PayPal.
